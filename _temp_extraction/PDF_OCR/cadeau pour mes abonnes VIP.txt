public struct MinuteData {
public double Open;
public double Close;
public double Volume;
public double BuyVolume;
public double SellVolume;

Close = open;
Volume = 0;

public double Volume;

Volume = volume;

private string currentMinuteKey;

private MinuteData currentMinuteData;
private List<CorridorData> corridorBuffer = new List<CorridorData>();
private double previousClose;
private double rSquared;
private bool isFirstMinute = true;
private const int corridorSize = 30;
protected override void OnStart()
{
currentMinuteKey = "";
isFirstMinute = true;
AddPlot(new PlotAttributes("R Squared", EColor.Blue, PlotStyle.Line));
}
protected override void OnTick(Tick tick)
{
if (tick.TickType != TickType.Trade) return;
string newMinuteKey = GetMinuteKey(tick);
if (newMinuteKey != currentMinuteKey)
{
if (!string.IsNullOrWhitespace(currentMinuteKey))
{
if (!isFirstMinute)
{
double priceChange = currentMinuteData.Close - previousClose;
double volume = currentMinuteData.Volume;
double delta = currentMinuteData.BuyVolume - currentMinuteData.SellVolume;
CorridorData newData = new CorridorData(priceChange, volume, delta);
corridorBuffer.Add(newData);

corridorBuffer.RemoveAt(0);

else

isFirstMinute = false;

previousClose = currentMinuteData.Close;
currentMinuteData = new MinuteData(tick.Bid);
}
else

currentMinuteData = new MinuteData(tick.Bid);
previousClose = tick.Bid;
isFirstMinute = false;
}
currentMinuteKey = newMinuteKey;

}
private string GetMinuteKey(Tick tick)

currentMinuteData.Close = tick.Bid;
currentMinuteData.Volume += tick.LastVolume;

currentMinuteData.BuyVolume += tick.LastVolume;

private double CalculateRSquared(List<CorridorData> data)
{
int n = data.Count;
double[] priceChanges = data.Select(d => d.PriceChange).ToArray();
double[] volumes = data.Select(d => d.Volume).ToArray();
double[] deltas = data.Select(d => d.Delta).ToArray();
double sum1 = n;
double sumV = volumes.Sum();
double sumD = deltas.Sum();
double sumVV = volumes.Select(v => v * v).Sum();
double sumVD = Enumerable.Zip(volumes, deltas).Select(pair => pair.First *
pair.Second).Sum();
double sumDD = deltas.Select(d => d * d).Sum();
double sumP = priceChanges.Sum();
double sumVP = Enumerable.Zip(volumes, priceChanges).Select(pair => pair.First *
pair.Second).Sum();
double sumDP = Enumerable.Zip(deltas, priceChanges).Select(pair => pair.First *
pair.Second).Sum();

double[,] XtX = new double[3, 3] {
{ sum1, sumV, sumD },
{ sumV, sumVV, sumVD },
{ sumD, sumVD, sumDD }
};
double[] XtY = new double[] { sumP, sumVP, sumDP };
double[,] invXtX = MatrixInverse(XtX);
double[] coefficients = MatrixMultiply(invXtX, XtY);
double[] predicted = new double[n];
for (int i = 0; i < n; i++)
{
predicted[i] = coefficients[0] + coefficients[1] * volumes[i] + coefficients[2] * deltas[i];
}
double sumResidualsSquared = priceChanges.Zip(predicted).Select(pair =>
Math.Pow(pair.First - pair.Second, 2)).Sum();
double meanP = sumP / n;
double sumTotalVariation = priceChanges.Select(p => Math.Pow(p - meanP, 2)).Sum();
if (sumTotalVariation == 0)
{
return double.NaN;
}
double rSquared = 1 - (sumResidualsSquared / sumTotalVariation);
return rSquared;
}
private double[,] MatrixInverse(double[,] matrix)
{
double a = matrix[0,0], b = matrix[0,1], c = matrix[0,2];
double d = matrix[1,0], e = matrix[1,1], f = matrix[1,2];
double g = matrix[2,0], h = matrix[2,1], i = matrix[2,2];
double determinant = a*(e*i - f*h) - b*(d*i - f*g) + c*(d*h - e*g);
if (determinant == 0)
{
throw new Exception("Singular matrix");
}
double invDet = 1 / determinant;
double[,] invMatrix = new double[3,3] {
{ (e*i - f*h)*invDet, (c*h - b*i)*invDet, (b*f - c*e)*invDet },
{ (f*g - d*i)*invDet, (a*i - c*g)*invDet, (c*d - a*f)*invDet },
{ (d*h - e*g)*invDet, (b*g - a*h)*invDet, (a*e - b*d)*invDet }
};
return invMatrix;
}

private double[] MatrixMultiply(double[,] matrix, double[] vector)
{
int rows = matrix.GetLength(0);
int cols = matrix.GetLength(1);
if (cols != vector.Length)
{
throw new Exception("Matrix and vector dimensions do not match");
}
double[] result = new double[rows];
for (int i = 0; i < rows; i++)
{
double sum = 0;
for (int j = 0; j < cols; j++)
{
sum += matrix[i,j] * vector[j];
}
result[i] = sum;
}
return result;
}

Predicting Expected Return
To predict the expected return for the next 30-minute corridor, the analysis suggests
using the current corridor’s R-squared and price change. The proposed method is:
● Calculate the total price change for the current corridor by summing all
minute-to-minute price changes (corridorBuffer.Sum(d => d.PriceChange)).
● Set the expected return for the next corridor as expected_return = R-squared *
total_price_change.
This approach assumes that higher predictability (high R-squared) means the current
corridor’s price movement is likely to persist, so the expected return for the next corridor
is proportional to the current corridor’s total price change, weighted by R-squared. If
R-squared is low, the expected return is closer to zero, reflecting market randomness.
Implementation in NinjaTrader
The implementation builds on the previous tick-based indicator code, modifying it to
include the expected return calculation and plotting. The key steps are:
1. Data Collection: Use tick data to track trades, determine buy/sell volumes for
delta, and aggregate data per minute (open, close, volume, delta).
2. Corridor Management: Maintain a buffer of the last 30 minutes’ data (price
change, volume, delta), updating it as new minutes complete.
3. R-squared Calculation: When the buffer has 30 entries, calculate R-squared
using multiple regression, as implemented in the previous code.
4. Expected Return Prediction: Calculate total price change for the corridor and
compute expected_return = R-squared * total_price_change.
5. Plotting: Plot the expected_return at the end of the current corridor, which
corresponds to the start of the next corridor, providing a real-time forecast.

using System;
using System.Linq;
namespace NinjaTrader.NinjaScript
{
[NinjaScriptIndicator]
public class PredictabilityIndicator : Indicator
{
public struct MinuteData {
public double Open;
public double Close;
public double Volume;

public double BuyVolume;
public double SellVolume;
public MinuteData(double open) {
Open = open;
Close = open;
Volume = 0;
BuyVolume = 0;
SellVolume = 0;
}
}
public struct CorridorData {
public double PriceChange;
public double Volume;
public double Delta;
public CorridorData(double priceChange, double volume, double delta) {
PriceChange = priceChange;
Volume = volume;
Delta = delta;
}
}
private string currentMinuteKey;
private MinuteData currentMinuteData;
private List<CorridorData> corridorBuffer = new List<CorridorData>();
private double previousClose;
private double rSquared;
private bool isFirstMinute = true;
private const int corridorSize = 30;
protected override void OnStart()
{
currentMinuteKey = "";
isFirstMinute = true;
AddPlot(new PlotAttributes("Expected Return", EColor.Green, PlotStyle.Line));
}
protected override void OnTick(Tick tick)
{
if (tick.TickType != TickType.Trade) return;
string newMinuteKey = GetMinuteKey(tick);
if (newMinuteKey != currentMinuteKey)
{
if (!string.IsNullOrWhitespace(currentMinuteKey))
{
if (!isFirstMinute)
{
double priceChange = currentMinuteData.Close - previousClose;

double volume = currentMinuteData.Volume;
double delta = currentMinuteData.BuyVolume - currentMinuteData.SellVolume;
CorridorData newData = new CorridorData(priceChange, volume, delta);
corridorBuffer.Add(newData);
if (corridorBuffer.Count > corridorSize)
{
corridorBuffer.RemoveAt(0);
}
if (corridorBuffer.Count == corridorSize)
{
rSquared = CalculateRSquared(corridorBuffer);
double total_price_change = corridorBuffer.Sum(d => d.PriceChange);
double expected_return = rSquared * total_price_change;
Plot(CurrentTime(), expected_return);
}
}
else
{
isFirstMinute = false;
}
previousClose = currentMinuteData.Close;
currentMinuteData = new MinuteData(tick.Bid);
}
else
{
currentMinuteData = new MinuteData(tick.Bid);
previousClose = tick.Bid;
isFirstMinute = false;
}
currentMinuteKey = newMinuteKey;
}
UpdateMinuteData(tick);
}
private string GetMinuteKey(Tick tick)
{
return tick.Time.ToString("yyyyMMddHHmm");
}
private void UpdateMinuteData(Tick tick)
{
currentMinuteData.Close = tick.Bid;
currentMinuteData.Volume += tick.LastVolume;
if (Math.Abs(tick.LastPrice - tick.Ask) < 1e-6)
{
currentMinuteData.BuyVolume += tick.LastVolume;

}
else if (Math.Abs(tick.LastPrice - tick.Bid) < 1e-6)
{
currentMinuteData.SellVolume += tick.LastVolume;
}
}
private double CalculateRSquared(List<CorridorData> data)
{
int n = data.Count;
double[] priceChanges = data.Select(d => d.PriceChange).ToArray();
double[] volumes = data.Select(d => d.Volume).ToArray();
double[] deltas = data.Select(d => d.Delta).ToArray();
double sum1 = n;
double sumV = volumes.Sum();
double sumD = deltas.Sum();
double sumVV = volumes.Select(v => v * v).Sum();
double sumVD = Enumerable.Zip(volumes, deltas).Select(pair => pair.First *
pair.Second).Sum();
double sumDD = deltas.Select(d => d * d).Sum();
double sumP = priceChanges.Sum();
double sumVP = Enumerable.Zip(volumes, priceChanges).Select(pair => pair.First *
pair.Second).Sum();
double sumDP = Enumerable.Zip(deltas, priceChanges).Select(pair => pair.First *
pair.Second).Sum();
double[,] XtX = new double[3, 3] {
{ sum1, sumV, sumD },
{ sumV, sumVV, sumVD },
{ sumD, sumVD, sumDD }
};
double[] XtY = new double[] { sumP, sumVP, sumDP };
double[,] invXtX = MatrixInverse(XtX);
double[] coefficients = MatrixMultiply(invXtX, XtY);
double[] predicted = new double[n];
for (int i = 0; i < n; i++)
{
predicted[i] = coefficients[0] + coefficients[1] * volumes[i] + coefficients[2] * deltas[i];
}
double sumResidualsSquared = priceChanges.Zip(predicted).Select(pair =>
Math.Pow(pair.First - pair.Second, 2)).Sum();
double meanP = sumP / n;
double sumTotalVariation = priceChanges.Select(p => Math.Pow(p - meanP, 2)).Sum();
if (sumTotalVariation == 0)
{
return double.NaN;

}
double rSquared = 1 - (sumResidualsSquared / sumTotalVariation);
return rSquared;
}
private double[,] MatrixInverse(double[,] matrix)
{
double a = matrix[0,0], b = matrix[0,1], c = matrix[0,2];
double d = matrix[1,0], e = matrix[1,1], f = matrix[1,2];
double g = matrix[2,0], h = matrix[2,1], i = matrix[2,2];
double determinant = a*(e*i - f*h) - b*(d*i - f*g) + c*(d*h - e*g);
if (determinant == 0)
{
throw new Exception("Singular matrix");
}
double invDet = 1 / determinant;
double[,] invMatrix = new double[3,3] {
{ (e*i - f*h)*invDet, (c*h - b*i)*invDet, (b*f - c*e)*invDet },
{ (f*g - d*i)*invDet, (a*i - c*g)*invDet, (c*d - a*f)*invDet },
{ (d*h - e*g)*invDet, (b*g - a*h)*invDet, (a*e - b*d)*invDet }
};
return invMatrix;
}
private double[] MatrixMultiply(double[,] matrix, double[] vector)
{
int rows = matrix.GetLength(0);
int cols = matrix.GetLength(1);
if (cols != vector.Length)
{
throw new Exception("Matrix and vector dimensions do not match");
}
double[] result = new double[rows];
for (int i = 0; i < rows; i++)
{
double sum = 0;
for (int j = 0; j < cols; j++)
{
sum += matrix[i,j] * vector[j];
}
result[i] = sum;
}
return result;
}
}

Practical Considerations and Limitations
Several factors must be considered:
● Data Feed Dependency: The indicator requires tick data with bid and ask prices
for accurate delta calculation. Ensure the data feed supports Level II data
(NinjaTrader Data Feeds).
● Performance: Real-time calculation of R-squared and expected return for each
minute may impact performance, especially with high-frequency data. Optimize
by handling matrix operations efficiently.
● Edge Cases: Handle minutes with no trades by using previous close prices, and
manage matrix singularity by returning NaN for R-squared, setting
expected_return to zero in such cases.
● Time Zone Handling: Ensure time tracking accounts for any time zone offsets,
using NinjaScript’s DateTime properties.
● Backtesting: The effectiveness of this approach requires backtesting to validate
whether predicted expected returns correlate with actual returns, as markets may
be efficient, and predictability might not always lead to profits.

Innovation and Trading Implications

The approach is innovative because it integrates a statistical predictability measure
(R-squared from multiple regression) with volume and order book data, which is less
common in traditional indicators. Literature on trading strategies based on predictability
measures, such as A Trading Approach to Testing for Predictability, supports the idea
that predictability can inform trading decisions, though its profitability depends on market
conditions and backtesting. The user’s method of predicting expected returns based on
current corridor predictability is novel, potentially offering an edge in short-term trading,
but its success requires rigorous testing to ensure it isn’t arbitraged away by market
efficiency.
